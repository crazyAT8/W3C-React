React ES6 Ternary Operator

Ternary Operator

  • The ternary operator is a simplified conditional operator like if/else.
  • Syntax: condition ? <expression if true> : <expression if false>
function renderApp() {
    document.getElementById("demo").innerHTML = "Welcome!";
}

function renderLogin() {
    document.getElementById("demo").innerHTML = "Please log in";
}

let authenticated = true; // change for it to switch

if (authenticated) {
    renderApp();
} else {
    renderLogin();
}

function renderApp2() {
    document.getElementById("demo").innerHTML = "Welcome!";
}

function renderLogin2() {
    document.getElementById("demo").innerHTML = "Please log in";
}

let authenticated2 = true; // change for it to switch

authenticated2 ? renderApp2() : renderLogin2();

// this is saying if the condition is true
// use the renderApp2() function
// if false then renderLogin2()
